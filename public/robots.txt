# /companies (no slug) is the private "My companies" dashboard — disallowed.
# /companies/<slug> is a public campaign page and must stay crawlable, hence
# the exact-match Disallow ($) plus an explicit Allow for the slug form.

# over the current page, and magic-link completion lands on "/" via a URL
# fragment, which is never sent to or seen by crawlers.

Disallow: /companies$
Allow: /companies/
